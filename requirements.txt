# VRPS-RunPod — trimmed Python deps for the HEADLESS OFFLINE pipeline.
#
# Dropped vs upstream pyproject (GUI / realtime / DLNA only):
#   PySide6-Essentials, pyinstaller, av, pymp4, pynvvideocodec, py-spy
# Kept: the ONNX + CuPy + TensorRT runtime that the offline 2D->3D/VR and
# matting (RVM / MatAnyone2 / SAM3) paths actually import.
#
# sm_120 / Blackwell (RTX PRO 4500) notes are in the Dockerfile header.

# --- core numerics / IO ---
numpy>=1.26,<2.1
opencv-python-headless>=4.10          # headless build: no GUI libGL needed
pillow>=10

# --- ONNX Runtime with CUDA + TensorRT execution providers ---
# onnxruntime-gpu provides CUDAExecutionProvider and TensorrtExecutionProvider.
# Do NOT also install plain onnxruntime; it shadows the GPU build.
# IMPORTANT: 1.27.0 dropped CUDA 12 (it's CUDA-13-only -> needs libcudart.so.13,
# which this 12.8 image does not have). Pin below 1.27 to keep the CUDA 12
# build. 1.26.x is the last CUDA-12 release line.
onnxruntime-gpu>=1.19,<1.27
onnx>=1.16

# --- CuPy GPU renderer (sm_120 native via bundled CUDA toolkit) ---
cupy-cuda12x[ctk]

# --- cuDNN 9 in-venv (ORT CUDA EP requires it; not in the toolkit extra) ---
nvidia-cudnn-cu12>=9.7,<10

# --- TensorRT engine build/cache ---
# MUST stay on the v10 branch: onnxruntime-gpu <1.27 links against
# libnvinfer.so.10. tensorrt-cu12 11.x ships libnvinfer.so.11 -> ORT's TensorRT
# EP fails to load and silently falls back to the (slower) CUDA EP.
tensorrt-cu12>=10.0,<11

# --- model fetching (heavy models: MatAnyone2, SAM3) ---
huggingface_hub>=0.25

# --- SAM3 / YOLO-World prepass (CLIP tokenizer lives in osam) ---
# WARNING: osam depends on plain `onnxruntime`, which would SHADOW
# onnxruntime-gpu and silently drop the CUDA/TensorRT providers (CPU-only).
# It is therefore installed WITHOUT deps in the Dockerfile via a dedicated step
# (see below), not here. Do not move `osam` into this file.
#
#   pip install --no-deps osam==0.4.0
#
# osam's only runtime needs that the offline path actually hits are numpy +
# pillow (already pinned above) for the CLIP tokenizer in
# offline/sam3_matanyone2.py. If a future code path needs more of osam, add the
# specific transitive dep here explicitly rather than letting osam pull
# onnxruntime.
